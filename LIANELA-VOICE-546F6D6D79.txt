        document.addEventListener('DOMContentLoaded', () => {
            const mobileMenuButton = document.getElementById('mobile-menu-button');

            mobileMenuButton.addEventListener('click', () => {
                mobileMenu.classList.toggle('hidden');
            });

            const header = document.getElementById('header');
            const headerInner = document.getElementById('header-inner');
            window.onscroll = function() {
                if (window.pageYOffset > 20) {
                    headerInner.classList.add('border-gray-200');
                } else {
                    headerInner.classList.remove('border-gray-200');
                }
            };
            
            // Context switcher
            const saBtn = document.getElementById('sa-btn');
            const globalBtn = document.getElementById('global-btn');
            const saContext = document.getElementById('sa-context');
            const globalContext = document.getElementById('global-context');
            
            saBtn.addEventListener('click', () => {
                saContext.classList.remove('hidden');
                globalContext.classList.add('hidden');
                saBtn.classList.add('text-black', 'border-black');
                saBtn.classList.remove('text-gray-400');
                globalBtn.classList.remove('text-black', 'border-black');
                globalBtn.classList.add('text-gray-400');
            });

            globalBtn.addEventListener('click', () => {
                saContext.classList.add('hidden');
                globalContext.classList.remove('hidden');
                globalBtn.classList.add('text-black', 'border-black');
                globalBtn.classList.remove('text-gray-400');
                saBtn.classList.remove('text-black', 'border-black');
                saBtn.classList.add('text-gray-400');
                initComparisonChart(); 
            });


            let comparisonChartInstance = null;
            function initComparisonChart() {
                if (comparisonChartInstance) {
                    comparisonChartInstance.destroy();
                }
                const ctx = document.getElementById('comparisonChart').getContext('2d');
                const chartData = {
                    labels: ['Task Breadth', 'Context Length', 'SA Legal Specificity', 'Novel Methodology', 'Commercial Potential'],
                    datasets: [{
                        label: 'Project Dark Matter',
                        data: [9, 10, 10, 8, 9],
                        backgroundColor: 'rgba(0, 0, 0, 0.05)',
                        borderColor: 'rgba(0, 0, 0, 1)',
                        borderWidth: 1.5,
                        pointBackgroundColor: 'rgba(0, 0, 0, 1)',
                        pointBorderColor: '#fff',
                        pointRadius: 4,
                    }, {
                        label: 'Typical Global Project',
                        data: [5, 4, 2, 5, 6],
                        backgroundColor: 'rgba(0, 0, 0, 0.02)',
                        borderColor: 'rgba(0, 0, 0, 0.4)',
                        borderWidth: 1.5,
                        pointBackgroundColor: 'rgba(0, 0, 0, 0.4)',
                        pointBorderColor: '#fff',
                        pointRadius: 4,
                    }]
                };

                comparisonChartInstance = new Chart(ctx, {
                    type: 'radar',
                    data: chartData,
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            legend: {
                                position: 'bottom',
                                labels: {
                                    color: '#000000',
                                    font: {
                                        family: "'Quicksand', sans-serif",
                                        weight: '500'
                                    },
                                    usePointStyle: true,
                                    boxWidth: 8
                                }
                            },
                        },
                        scales: {
                            r: {
                                beginAtZero: true,
                                max: 10,
                                pointLabels: {
                                    color: '#000000',
                                    font: {
                                        family: "'Quicksand', sans-serif",
                                        size: 12,
                                        weight: '500'
                                    }
                                },
                                grid: { color: '#E5E7EB' },
                                angleLines: { color: '#E5E7EB' },
                                ticks: {
                                   display: false,
                                   stepSize: 2
                                }
                            }
                        }
                    }
                });
            }
            // Trigger chart render if the global tab is visible on load (for cases where user might reload on that tab)
            if(!globalContext.classList.contains('hidden')){
                initComparisonChart();
            }
        });
